#include<iostream>
using namespace std;
class matrix
{
        int mat[20][20],row,col;
        public:matrix(){}
        matrix(int r,int c){
                row=r;
                col=c;
        }

        void getdata()
               {
                       cout<<"Enter number of rows: ";
                       cin>>row;
                       cout<<"\nEnter number of columns: ";
                       cin>>col;
                       cout<<"\nEnter matrix elements: ";
                       for(int i=0;i<row;i++)
                               for(int j=0;j<col;j++)
                                       cin>>mat[i][j];
               }
               matrix operator +(matrix m)
               {
                       matrix res;
                       res.row=m.row;
                       res.col=m.col;
                       for(int i=0;i<m.row;i++)
                               for(int j=0;j<m.col;j++)
                                       res.mat[i][j]=mat[i][j]+m.mat[i][j];
                       return res;
               }
               matrix operator -(matrix m)
               {
                       matrix res;
                        res.row=m.row;
                        res.col=m.col;
                        for(int i=0;i<m.row;i++)                                                                                     for(int j=0;j<m.col;j++)
                                                                      res.mat[i][j]=mat[i][j]-m.mat[i][j];
                      return res;
               }
                friend bool operator==(matrix ,matrix );
                void display()
                {
                        cout<<"\nResult is: \n";
                        for(int i=0;i<row;i++)
                                for(int j=0;j<col;j++)
                                        cout<<mat[i][j]<<" ";
                }
};
bool operator==(matrix m1,matrix m2)
{
        if(m1.row==m2.row && m1.col==m2.col)
                return true;
        else
                return false;
}
int main()
{
        system("clear");
        matrix  m1,m2,m3,m4;
        m1.getdata();
        m2.getdata();
        if(m1==m2){
                m3=m1+m2;
                cout<<"\nSum is: ";
                m3.display();
                m4=m1-m2;
                cout<<"\nDifference is: ";
                m4.display();
        }else
                cout<<"\nMatrix is not compatible ";
        return 0;
}
